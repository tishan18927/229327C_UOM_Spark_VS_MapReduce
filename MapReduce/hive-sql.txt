SELECT Year, AVG((CASE WHEN ArrDelay = 0 THEN 0 ELSE LateAircraftDelay / ArrDelay END) * 100) AS AvgLateAircraftDelayPercentage FROM delay_flights WHERE Year >= 2003 AND Year <= 2010 GROUP BY Year;
